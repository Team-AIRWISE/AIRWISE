import sqlite3
import datetime



# Connect to the SQLite database
conn = sqlite3.connect("py.db")



# Create a cursor object
cur = conn.cursor()



# Define the SQL command to create the table
command1 = """CREATE TABLE IF NOT EXISTS
loggs(Time TEXT PRIMARY KEY, Temperature REAL, humidity REAL, eco2 REAL, aqs REAL, aqp REAL, pressure REAL)"""



# Execute the SQL command to create the table
cur.execute(command1)



# Define values to insert into the table
currentTime = str(datetime.datetime.now())
temperature = 25  # Example temperature value
humidity = 60  # Example humidity value
co2 = 400  # Example CO2 value
aqi = 50  # Example AQI value
aqp = 100  # Example AQP value
pressure = 1013.25  # Example pressure value



# Execute the SQL query to insert values into the table
cur.execute("INSERT INTO loggs (Time, Temperature, humidity, eco2, aqs, aqp, pressure) VALUES (?, ?, ?, ?, ?, ?, ?)",
            (currentTime, temperature, humidity, co2, aqi, aqp, pressure))



# Commit the transaction to save the changes
conn.commit()



# Close the connection
conn.close()
